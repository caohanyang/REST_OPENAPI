054  // A comparator that sorts annotations by length; longest first
055  private static Comparator<Annotation> lengthComparator =
056    new Comparator<Annotation>() {
057      @Override
058      public int compare(Annotation a1, Annotation a2) {
059        long l1 = a1.getEndNode().getOffset() - a1.getStartNode().getOffset();
060        long l2 = a2.getEndNode().getOffset() - a2.getStartNode().getOffset();
061
062        return (int)(l2 - l1);
063      }
064    };

120  @Override
121  public Resource init() throws ResourceInstantiationException {
122
123    // load and configure the hidden gazetteer
124    FeatureMap hidden = Factory.newFeatureMap();
125    Gate.setHiddenAttribute(hidden, true);
126
127    FeatureMap params = Factory.newFeatureMap();
128    params.put("listsURL", gazURL);
129    params.put("caseSensitive", Boolean.FALSE);
130
131    params.put("longestMatchOnly", Boolean.FALSE);
132    params.put("wholeWordsOnly", Boolean.FALSE);
133
134    if(gaz == null) {
135      gaz =
136        (LanguageAnalyser)Factory.createResource(
137          "gate.creole.gazetteer.DefaultGazetteer", params, hidden,
138          "Hashtag Tokenizer Gazetteer");
139    } else {
140      gaz.setParameterValues(params);
141      gaz.reInit();
142    }
143
144    return this;
145  }

147  @Override
148  public void execute() throws ExecutionException {
149    // reset interrupt flag
150    interrupted = false;
151    AnnotationSet inputAS = document.getAnnotations(inputASName);
152    AnnotationSet outputAS = document.getAnnotations(outputASName);
153
154    FeatureMap features = Factory.newFeatureMap();
155
156    long startTime = System.currentTimeMillis();
157    fireStatusChanged("Tokenizing Hashtags: " + document.getName());
158    fireProgressChanged(0);
159    int count = 0;
160    
161    // get all the lookups we are going to use for decomposition...
162    AnnotationSet lookups = new AnnotationSetImpl(document);
163
164    try {
165      // run the gazetteer to produce the HashtagLookup annotations
166      gaz.setParameterValue("annotationSetName", inputASName);
167      gaz.setDocument(document);
168      gaz.execute();
169
170      // get all the hashtags
171      AnnotationSet hashtags = inputAS.get("Hashtag");
172
173      for(Annotation hashtag : inputAS.get("Hashtag")) {
174        // for each hashtag in the document...
175
176        AnnotationSet contained = inputAS.getContained(hashtag.getStartNode().getOffset(), hashtag.getEndNode().getOffset());
177        
178        //clear away any left overs from previous tags
179        lookups.clear();
180
181        // which are the HashtagLookup
182        lookups.addAll(contained.get("HashtagLookup"));
183
184        // any other Lookups the user has generated
185        lookups.addAll(contained.get("Lookup"));
186
187        // and any number tokens
188        features = Factory.newFeatureMap();
189        features.put("kind", "number");
190        lookups.addAll(contained.get("Token", features));
191
192        // the _ appears to be allowed so add them as well
193        features = Factory.newFeatureMap();
194        features.put("string", "_");
195        lookups.addAll(contained.get("Token", features));
196        
197        if(isInterrupted()) { throw new ExecutionInterruptedException(
198          "The execution of the hashtag tokenizer has been abruptly interrupted!"); }
199
200        // this will hold the best we have seen so far
201        List<List<Annotation>> fewestTokens = new ArrayList<List<Annotation>>();
202
203        // get all the lookups that start at the beginning of the hashtag
204        List<Annotation> start =
205          sort(getAnnotationsAtOffset(lookups, hashtag.getStartNode()
206            .getOffset() + 1));
207
208        for(Annotation a : start) {
209          // for each lookup search for a valid tokenization
210          List<List<Annotation>> found =
211            search(lookups, hashtag.getEndNode().getOffset(), a);
212
213          if(found != null) {
214            // if we found a contender and it's the best so far store it
215            if(fewestTokens.isEmpty()) {
216              fewestTokens.addAll(found);
217            } else if(found.get(0).size() == fewestTokens.get(0).size()) {
218              fewestTokens.addAll(found);
219            } else if(found.get(0).size() < fewestTokens.get(0).size()) {
220              fewestTokens.clear();
221              fewestTokens.addAll(found);
222            }
223          }
224        }
225
226        if(debug && fewestTokens.size() > 1) {
227          System.out.println(stringFor(document, hashtag));
228          display(fewestTokens);
229        }
230
231        if(fewestTokens.isEmpty()) {
232          // if we didn't find any sensible tokenizations then let's see if the
233          // hashtag is mized case
234
235          String tagText = stringFor(document, hashtag).substring(1);
236          if("mixedCaps".equals(getTokenType(tagText)[1])) {
237            // if we have a mixed case hahstag then let's assume it is
238            // CamelCased and split it accordingly
239
240            // TODO think about camel case which includes numbers
241
242            // a list to hold the tokens
243            List<Annotation> found = new ArrayList<Annotation>();
244
245            // start looking for token breaks aftert the initial #
246            long begin = hashtag.getStartNode().getOffset() + 1;
247
248            for(String token : tagText
249              .split("((?<=[a-z])(?=[A-Z]))|((?<=[A-Z]{2,})(?=[a-z]))")) {
250              // split the token at the case changes...
251
252              // create the annotation in the Lookup set and add it to the found
253              // list
254              found.add(lookups.get(lookups.add(begin,
255                (begin += token.length()), "CamelToken",
256                Factory.newFeatureMap())));
257            }
258
259            // record the tokenization so we can process it later
260            fewestTokens.add(found);
261          }
262
263        }
264
265        if(!fewestTokens.isEmpty()) {
266          // if we found a valid tokenization then...
267
268          // remove any existing Token annotations
269          inputAS.removeAll(inputAS.get("Token").getContained(
270            hashtag.getStartNode().getOffset(),
271            hashtag.getEndNode().getOffset()));
272
273          // create a punctuation Token over the #
274          features = Factory.newFeatureMap();
275          features.put("string", "#");
276          features.put("length", "1");
277          features.put("kind", "punctuation");
278          outputAS.add(hashtag.getStartNode().getOffset(), hashtag
279            .getStartNode().getOffset() + 1, "Token", features);
280
281          // let's assume that the first one we found is best
282          int prefered = 0;
283
284          for(int i = 0; i < fewestTokens.size(); ++i) {
285            // check those we have found and skip over any that contain single
286            // letter or mixed case words
287
288            boolean okay = true;
289            for(Annotation a : fewestTokens.get(i)) {
290              // single letter words are not great
291              if(a.getEndNode().getOffset() - a.getStartNode().getOffset() == 1)
292                okay = false;
293            }
294
295            if(okay) {
296              // if it contains neither a single letter word or a mixed case
297              // word then we should definitely prefer this one
298              prefered = i;
299              break;
300            }
301          }
302
303          for(Annotation a : fewestTokens.get(prefered)) {
304            // for each new token...
305
306            // find where it starts/ends and its length
307            long startOffset = a.getStartNode().getOffset();
308            long endOffset = a.getEndNode().getOffset();
309            String length = Long.toString(endOffset - startOffset);
310
311            // get the actual text
312            String string = stringFor(document, a);
313
314            // work out what kind of token it is and if it is a word
315            // what its orthography is
316            String[] tokenType = getTokenType(string);
317            String kind = tokenType[0];
318            String orth = tokenType[1];
319
320            // create the new Token annotation
321            features = Factory.newFeatureMap();
322            features.put("string", string);
323            features.put("length", length);
324            features.put("kind", kind);
325            if(orth != null) features.put("orth", orth);
326            outputAS.add(startOffset, endOffset, "Token", features);
327
328            if(debug) {
329              // for debug purposes add a matching set of HashtagToken
330              // annotations
331              features = Factory.newFeatureMap();
332              features.put("string", string);
333              features.put("length", length);
334              features.put("kind", kind);
335              if(orth != null) features.put("orth", orth);
336              outputAS.add(startOffset, endOffset, "HashtagToken", features);
337            }
338          }
339        } else if(debug) {
340          System.err.println(stringFor(document, hashtag));
341          
342          AnnotationSet tokens = inputAS.get("Token").getContained(
343            hashtag.getStartNode().getOffset()+1,
344            hashtag.getEndNode().getOffset());
345          
346          for (Annotation token : tokens) {
347            
348            features = Factory.newFeatureMap();
349            features.putAll(token.getFeatures());
350            outputAS.add(token.getStartNode().getOffset(), token.getEndNode().getOffset(), "HashtagToken", features);
351          }
352        }
353
354        fireProgressChanged(count++ * 100 / hashtags.size());
355      }
356
357      fireProcessFinished();
358      fireStatusChanged("Hashtags in " +
359        document.getName() +
360        " tokenized in " +
361        NumberFormat.getInstance().format(
362          (double)(System.currentTimeMillis() - startTime) / 1000) +
363        " seconds!");
364    } catch(InvalidOffsetException e) {
365      throw new ExecutionException(e);
366    } catch(ResourceInstantiationException e) {
367      throw new ExecutionException(e);
368    } finally {
369      gaz.setDocument(null);
370
371      if(!debug) {
372        inputAS.removeAll(inputAS.get("HashtagLookup"));
373      }
374    }
375  }

382  private static String[] getTokenType(String string) {
383    String kind = "symbol";
384    String orth = null;
385    if(isAlpha(string)) {
386      kind = "word";
387
388      if(isAllLowerCase(string))
389        orth = "lowercase";
390      else if(isAllUpperCase(string))
391        orth = "allCaps";
392      else if(isAllUpperCase(string.substring(0, 1)) &&
393        isAllLowerCase(string.substring(1)))
394        orth = "upperInitial";
395      else orth = "mixedCaps";
396    } else if(isNumeric(string)) {
397      kind = "number";
398    } else if(string.matches("(\\p{Punct})+")) {
399      kind = "punctuation";
400    }
401
402    return new String[]{kind, orth};
403  }

405  private void display(List<List<Annotation>> found) {
406    for(List<Annotation> tokens : found) {
407      System.out.print("   ");
408      for(Annotation token : tokens) {
409        System.out.print(stringFor(document, token) + " ");
410      }
411      System.out.print("\n");
412    }
413  }

415  /**
416   * Depth first search through a set of annotations to find a contiguous set
417   * for a given character range.
418   **/
419  private List<List<Annotation>> search(AnnotationSet lookups, Long endOffset,
420                                        Annotation token)
421    throws InvalidOffsetException {
422
423    if("mixedCaps"
424      .equals(getTokenType(stringFor(lookups.getDocument(), token))[1]))
425      return null;
426
427    List<List<Annotation>> shortest = new ArrayList<List<Annotation>>();
428
429    if(token.getEndNode().getOffset().equals(endOffset)) {
430      // if the token we are starting from ends at the right place then we
431      // can stop and just return the single token as the result
432      List<Annotation> found = new ArrayList<Annotation>();
433      found.add(token);
434
435      shortest.add(found);
436      return shortest;
437    }
438
439    if(endOffset - token.getEndNode().getOffset() > 1) {
440      // if there are two or more characters after this token then...
441
442      // get the rest of the text of the hashtag
443      String rest =
444        lookups.getDocument().getContent()
445          .getContent(token.getEndNode().getOffset() - 1, endOffset).toString();
446
447      if(rest.substring(1).matches(rest.substring(0, 1) + "+")) {
448        // if the rest of the hashtag is just the same as the last letter of
449        // this token then someone has just lent on the keyboard for emphasis
450
451        // so... extend the current token to include the rest of the hashtag
452        Annotation newToken =
453          lookups.get(lookups.add(token.getStartNode().getOffset(), endOffset,
454            "HashtagLookup", Factory.newFeatureMap()));
455
456        // return this extended token
457        List<Annotation> found = new ArrayList<Annotation>();
458        found.add(newToken);
459
460        shortest.add(found);
461        return shortest;
462      }
463    }
464
465    // get all possibilities that start after the current annotation that
466    // could be used to cover the next bit of the document
467    List<Annotation> next =
468      sort(getAnnotationsAtOffset(lookups, token.getEndNode().getOffset()));
469
470    // if there aren't any annotations we can use then return null as we can
471    // never fully cover the range from this point
472    if(next == null || next.isEmpty()) return null;
473
474    for(Annotation a : next) {
475      // use each possible annotation to start a new search
476      List<List<Annotation>> part = search(lookups, endOffset, a);
477
478      if(part != null) {
479        if(shortest.isEmpty()) {
480          shortest.addAll(part);
481        } else if(part.get(0).size() == shortest.get(0).size()) {
482          shortest.addAll(part);
483        } else if(part.get(0).size() < shortest.get(0).size()) {
484          shortest.clear();
485          shortest.addAll(part);
486        }
487      }
488    }
489
490    // if we didn't find a match then return null
491    if(shortest.isEmpty()) return null;
492
493    // add the token we started from to the beginning of the match
494    for(List<Annotation> found : shortest) {
495      found.add(0, token);
496    }
497
498    // return the match we found
499    return shortest;
500  }

502  /**
503   * Returns a list in which the Annotations are sorted by length, longest
504   * first.
505   **/
506  private List<Annotation> sort(AnnotationSet annotations) {
507
508    List<Annotation> sorted = new ArrayList<Annotation>();
509
510    if(annotations == null || annotations.isEmpty()) return sorted;
511
512    sorted.addAll(annotations);
513    Collections.sort(sorted, lengthComparator);
514
515    // TODO filter out annotations with the same span
516
517    List<Annotation> filtered = new ArrayList<Annotation>();
518    long length = -1;
519    for(Annotation a : sorted) {
520      long al = a.getEndNode().getOffset() - a.getStartNode().getOffset();
521      if(length == -1 || al != length) {
522        filtered.add(a);
523        length = al;
524      }
525    }
526
527    return filtered;
528  }
